139   def self.parse_argv(command_line, all_flags, usage)
 
140     if command_line.class != Array

143

148       if arg[0].chr == "-"

150         arg = arg[1, arg.length] if arg[0].chr == "-" # to handle --arg
 
151         arg_found = arg
 
152         if !all_flags.include?(arg)
 
153           raise BadCommandLineArgException.new("The flag #{arg} cannot be used here.")
 
154         end
