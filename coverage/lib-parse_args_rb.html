144     if command_line.class != Array

146     end

151     command_line.each { |arg|
 
152       if arg[0].chr == "-"
 
153         arg = arg[1, arg.length]
 
154         arg = arg[1, arg.length] if arg[0].chr == "-" # to handle --arg
 
155         arg_found = arg
 
156         if !all_flags.include?(arg)
 
157           raise BadCommandLineArgException.new("The flag #{arg} cannot be used here.")
 
158         end

160         arg_hash[arg] = "NO ARG"
 
161       elsif !arg_found.nil?

166       end
