162     # :call-seq:
 
163     #   and_return(value)
 
164     #   and_return(value, value, ...)
 
165     #   and_return { |*args| code }
 
166     #
 
167     # Declare that the method returns a particular value (when the
 
168     # argument list is matched).
 
169     #
 
170     # * If a single value is given, it will be returned for all matching
 
171     #   calls.
 
172     # * If multiple values are given, each value will be returned in turn for
 
173     #   each successive call.  If the number of matching calls is greater
 
174     #   than the number of values, the last value will be returned for
 
175     #   the extra matching calls.
 
176     # * If a block is given, it is evaluated on each call and its
 
177     #   value is returned.
 
178     #
 
179     # For example:
 
180     #
 
181     #  mock.should_receive(:f).returns(12)   # returns 12
 
182     #
 
183     #  mock.should_receive(:f).with(String). # returns an
 
184     #    returns { |str| str.upcase }        # upcased string
 
185     #
 
186     # +returns+ is an alias for +and_return+.
 
187     #
 
188     def and_return(*args, &block)
 
189       if block_given?
 
190         @return_queue << block
 
191       else
 
192         args.each do |arg|
 
193           @return_queue << lambda { |*a| arg }
 
194         end
 
195       end
 
196       self
 
197     end

332     # Declare that the given method must be called in order.  All
 
333     # ordered method calls must be received in the order specified by
 
334     # the ordering of the +should_receive+ messages.  Receiving a
 
335     # methods out of the specified order will cause a test failure.
 
336     #
 
337     # If the user needs more fine control over ordering
 
338     # (e.g. specifying that a group of messages may be received in any
 
339     # order as long as they all come after another group of messages),
 
340     # a _group_ _name_ may be specified in the +ordered+ calls.  All
 
341     # messages within the same group may be received in any order.
 
342     #
 
343     # For example, in the following, messages +flip+ and +flop+ may be
 
344     # received in any order (because they are in the same group), but
 
345     # must occur strictly after +start+ but before +end+.  The message
 
346     # +any_time+ may be received at any time because it is not
 
347     # ordered.
 
348     #
 
349     #    m = FlexMock.new
 
350     #    m.should_receive(:any_time)
 
351     #    m.should_receive(:start).ordered
 
352     #    m.should_receive(:flip).ordered(:flip_flop_group)
 
353     #    m.should_receive(:flop).ordered(:flip_flop_group)
 
354     #    m.should_receive(:end).ordered
 
355     #

357       if @globally
 
358         @global_order_number = define_ordered(group_name, @mock.flexmock_container)

361       end
 
362       @globally = false

364     end
