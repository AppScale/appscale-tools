231     def braindead_factorize(num)
 
232       return [0] if num == 0
 
233       return [-1] + braindead_factorize(-num) if num < 0
 
234       factors = []
 
235       while num % 2 == 0
 
236         factors << 2
 
237         num /= 2
 
238       end
 
239       size = num
 
240       n = 3
 
241       max = Math.sqrt(num)
 
242       while n <= max && n <= size
 
243         while size % n == 0
 
244           size /= n
 
245           factors << n
 
246         end
 
247         n += 2
 
248       end
