58     # Create a mocking object in the FlexMock framework.  The +flexmock+
 
59     # method has a number of options available, depending on just what kind of
 
60     # mocking object your require.  Mocks created via +flexmock+ will be
 
61     # automatically verify during the teardown phase of your test framework.
 
62     #
 
63     # :call-seq:
 
64     #   flexmock() { |mock| ... }
 
65     #   flexmock(name) { |mock| ... }
 
66     #   flexmock(expect_hash) { |mock| ... }
 
67     #   flexmock(name, expect_hash) { |mock| ... }
 
68     #   flexmock(real_object) { |mock| ... }
 
69     #   flexmock(real_object, name) { |mock| ... }
 
70     #   flexmock(real_object, name, expect_hash) { |mock| ... }
 
71     #   flexmock(:base, string, name, expect_hash) { |mock| ... }
 
72     #
 
73     # <b>Note:</b> A plain flexmock() call without a block will return the
 
74     # mock object (the object that interprets <tt>should_receive</tt> and its
 
75     # brethern). A flexmock() call that _includes_ a block will return the
 
76     # domain objects (the object that will interpret domain messages) since
 
77     # the mock will be passed to the block for configuration. With regular
 
78     # mocks, this distinction is unimportant because the mock object and the
 
79     # domain object are the same object.  However, with partial mocks, the
 
80     # mock object is separation from the domain object.  Keep that distinciton
 
81     # in mind.
 
82     #
 
83     # name ::
 
84     #   Name of the mock object.  If no name is given, "unknown" is used for
 
85     #   full mocks and "flexmock(<em>real_object</em>)" is used for partial
 
86     #   mocks.
 
87     #
 
88     # expect_hash ::
 
89     #   Hash table of method names and values.  Each method/value pair is
 
90     #   used to setup a simple expectation so that if the mock object
 
91     #   receives a message matching an entry in the table, it returns
 
92     #   the associated value.  No argument our call count constraints are
 
93     #   added.  Using an expect_hash is identical to calling:
 
94     #
 
95     #       mock.should_receive(method_name).and_return(value)
 
96     #
 
97     #   for each of the method/value pairs in the hash.
 
98     #
 
99     # real_object ::
 
100     #   If a real object is given, then a partial mock is constructed
 
101     #   using the real_object as a base. Partial mocks (formally referred
 
102     #   to as stubs) behave as a mock object when an expectation is matched,
 
103     #   and otherwise will behave like the original object.  This is useful
 
104     #   when you want to use a real object for testing, but need to mock out
 
105     #   just one or two methods.
 
106     #
 
107     # :base ::
 
108     #   Forces the following argument to be used as the base of a
 
109     #   partial mock object.  This explicit tag is only needed if you
 
110     #   want to use a string or a symbol as the mock base (string and
 
111     #   symbols would normally be interpretted as the mock name).
 
112     #
 
113     # &block ::
 
114     #   If a block is given, then the mock object is passed to the block and
 
115     #   expectations may be configured within the block.  When a block is given
 
116     #   for a partial mock, flexmock will return the domain object rather than
 
117     #   the mock object.
 
118     #
 
119     def flexmock(*args)
 
120       name = nil
 
121       quick_defs = {}
 
122       domain_obj = nil
 
123       safe_mode = false
 
124       model_class = nil
 
125       while ! args.empty?
 
126         case args.first
 
127         when :base, :safe
 
128           safe_mode = (args.shift == :safe)
 
129           domain_obj = args.shift
 
130         when :model
 
131           args.shift
 
132           model_class = args.shift
 
133         when String, Symbol
 
134           name = args.shift.to_s
 
135         when Hash
 
136           quick_defs = args.shift
 
137         else
 
138           domain_obj = args.shift
 
139         end
 
140       end
 
141       raise UsageError, "a block is required in safe mode" if safe_mode && ! block_given?
 
142 
143       if domain_obj
 
144         mock = ContainerHelper.make_partial_proxy(self, domain_obj, name, safe_mode)
 
145         result = domain_obj
 
146       elsif model_class
 
147         id = ContainerHelper.next_id
 
148         result = mock = FlexMock.new("#{model_class}_#{id}", self)
 
149       else
 
150         result = mock = FlexMock.new(name || "unknown", self)
 
151       end
 
152       mock.should_receive(quick_defs)
 
153       yield(mock) if block_given?
 
154       flexmock_remember(mock)
 
155       ContainerHelper.add_model_methods(mock, model_class, id) if model_class
 
156       result
 
157     end

263     # Build the chain of mocks for demeter style mocking.
 
264     #
 
265     # Warning: Nasty code ahead.
 
266     #
 
267     # This method builds a chain of mocks to support demeter style
 
268     # mocking.  Given a mock chain of "first.second.third.last", we
 
269     # must build a chain of mock methods that return the next mock in
 
270     # the chain.  The expectation for the last method of the chain is
 
271     # returned as the result of the method.
 
272     #
 
273     # Things to consider:
 
274     #
 
275     # (1) The expectation for the "first" method must be created by
 
276     # the proper mechanism, which is supplied by the block parameter
 
277     # "block".  In other words, first expectation is created by
 
278     # calling the block.  (This allows us to create expectations on
 
279     # both pure mocks and partial mocks, with the block handling the
 
280     # details).
 
281     #
 
282     # (2) Although the first mock is arbitrary, the remaining mocks in
 
283     # the chain will always be pure mocks created specifically for
 
284     # this purpose.
 
285     #
 
286     # (3) The expectations for all methods but the last in the chain
 
287     # will be setup to expect no parameters and to return the next
 
288     # mock in the chain.
 
289     #
 
290     # (4) It could very well be the case that several demeter chains
 
291     # will be defined on a single mock object, and those chains could
 
292     # share some of the same methods (e.g. "mock.one.two.read" and
 
293     # "mock.one.two.write" both share the methods "one" and "two").
 
294     # It is important that the shared methods return the same mocks in
 
295     # both chains.
 
296     #
 
297     def build_demeter_chain(mock, arg, &block)
 
298       container = mock.flexmock_container
 
299       names = arg.to_s.split('.')
 
300       check_method_names(names)
 
301       exp = nil
 
302       next_exp = lambda { |n| block.call(n) }
 
303       loop do
 
304         method_name = names.shift.to_sym
 
305         exp = mock.flexmock_find_expectation(method_name)
 
306         need_new_exp = exp.nil? || names.empty?
 
307         exp = next_exp.call(method_name) if need_new_exp
 
308         break if names.empty?
 
309         if need_new_exp
 
310           mock = container.flexmock("demeter_#{method_name}")
 
311           exp.with_no_args.and_return(mock)
 
312         else
 
313           mock = exp._return_value([])
 
314         end
 
315         check_proper_mock(mock, method_name)
 
316         next_exp = lambda { |n| mock.should_receive(n) }
 
317       end
 
318       exp
 
319     end
