def initialize id, cloud, roles=[]
  # For Xen/KVM id is the public ip address
  # For clouds, id is node-X since the ip is not known
  @id = id
  @cloud = cloud
  @roles = roles.map { |r| r.to_sym }

  expand_roles
end
